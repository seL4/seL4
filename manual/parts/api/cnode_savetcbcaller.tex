\apidoc
{cnode_savecaller}
{CNode  -  Save TCB Caller}
{Save the reply capability from the last time the target thread was called in the given CSpace so that it can be invoked later}
{static inline int seL4\_CNode\_SaveTCBCaller}
{
\param{seL4\_CNode}{\_service}{
    CPTR to the CNode at the root of the CSpace where the capability is to be
    saved. Must be at a depth of 32.
}
\param{seL4\_Word}{index}{
    CPTR to the slot in which to save the capability. Resolved from the root of
    the \_service parameter.
}
\param{uint8\_t}{depth}{
    Number of bits of index to resolve to find the slot being targeted.
}
\param{seL4\_CPtr}{tcb}{
    A capability to the target thread. 
}
}
{\errorenumdesc}
{See \autoref{sec:cnode-ops}.}
